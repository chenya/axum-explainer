#[derive(Debug, thiserror::Error)] pub enum AppError { #[error("not found: id={id}")] NotFound { id: i32 }, #[error("database error: {0}")] Database(#[from] sqlx::Error), } impl IntoResponse for AppError { fn into_response(self) -> Response { match &self { AppError::NotFound { .. } => (StatusCode::NOT_FOUND, self.to_string()), AppError::Database(_) => (StatusCode::INTERNAL_SERVER_ERROR, self.to_string()), }.into_response() } } // Now every handler can return Result<T, AppError> async fn get_movie( State(state): State<AppState>, Path(id): Path<i32>, ) -> Result<Json<Movie>, AppError> { sqlx::query_as!(Movie, "SELECT * FROM movie WHERE id = $1", id) .fetch_optional(&state.db).await .map_err(AppError::Database)? .map(Json) .ok_or(AppError::NotFound { id }) }
